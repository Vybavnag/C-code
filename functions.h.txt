void printVector(std::vector<std::string> letter){
    for (int i = 0; i < letter.size(); i++){
        std::cout << letter[i];
        if (i < letter.size() - 1){
            std::cout << " ";
        }
    }
    std::cout << std::endl;
}

     int pos = data.find(toSearch);

bool ifLetter(std::string word,std::string letter){
    if(word.find(letter)>word.size()){
        return false;
    }
    else{
        return true;
    }
    
}
bool compareString(std::string word1[], std::string word2,int o){
    for(int i=0; i<o;i++){
        std::string temp=word1[i];
        std::string temp2=word2.substr(i,1);
        if (temp!=temp2){
            return false;
        }
    }
    return true;
}
