#ifndef LIST_UTILS_H
#define LIST_UTILS_H

namespace ucm{

float addUp(float arr[],int size){
    float total=0;
    for(int i=0; i<size;i++){
        total=arr[i]+total;
    }
    return total;
}

float average(float arr[], int size){
    float total=0;
    for(int i=0; i<size;i++){
        total=arr[i]+total;
    }
    float answer= total/size;
    return answer;
}

float min(float arr[],int size){
    if(size==1){
        return arr[0];
    }
    float minimum=arr[size-2];
    for(int i=0; i<size;i++){
        if(minimum>arr[i]){
            minimum=arr[i];
        }
    }
    return minimum;
}
float max(float arr[],int size){
    float maximum=0;
    for(int i=0; i<size;i++){
        if(maximum<arr[i]){
            maximum=arr[i];
        }
    }
    return maximum;
}
}


#endif
